  const observer = new IntersectionObserver((entries) => {

  }, { threshold: 0.1 });
  document.querySelectorAll('.fade-in').forEach(el => observer.observe(el));

  // Publication tabs
  function switchTab(name, btn) {
    document.querySelectorAll('.pub-tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.pub-panel').forEach(p => p.classList.remove('active'));
    btn.classList.add('active');
    document.getElementById('tab-' + name).classList.add('active');
  }

  // Toggle show more publications
  var pubState = { articles: false, chapters: false };
  function togglePubs(type) {
    var more = document.getElementById(type + '-more');
    var btn = document.getElementById(type + '-toggle');
    pubState[type] = !pubState[type];
    if (pubState[type]) {
      more.style.display = 'block';
      btn.textContent = 'Show Less ↑';
    } else {
      more.style.display = 'none';
      btn.textContent = type === 'articles' ? 'Show All 16 Articles ↓' : 'Show All 19 Chapters ↓';
    }
  }
